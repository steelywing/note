<!-- config.xml --><RIBCL VERSION="2.1"> <LOGIN USER_LOGIN="Administrator" PASSWORD="password">  <USER_INFO MODE="write">    <MOD_USER USER_LOGIN="Administrator">    <PASSWORD VALUE="<new password>"/>    </MOD_USER>  </USER_INFO> </LOGIN></RIBCL>

ESXi

Reset license#

#!/bin/shrm /etc/vmware/license.cfgcp /etc/vmware/.#license.cfg /etc/vmware/license.cfg/etc/init.d/vpxa restart

Cron#

Default /var/spool/cron/crontabs/root is read only, use :w! in vi to save.

kill $(cat /var/run/crond.pid)vi /var/spool/cron/crontabs/rootcrond

ESXi will overwrite cron file on boot, change /etc/rc.local.d/local.sh to update cron file

# /etc/rc.local.d/local.shkill $(cat /var/run/crond.pid)echo "* * * * * <cron job>" >> /var/spool/cron/crontabs/rootcrond
